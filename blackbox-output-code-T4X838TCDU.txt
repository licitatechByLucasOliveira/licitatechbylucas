    <script>
        let cart = [];
        let totalVendas = 0;
        let numVendas = 0;
        let userLogged = '';
        let userType = '';

        const modalColab = document.getElementById('modal-colaborador');
        const modalDono = document.getElementById('modal-dono');
        const modalCart = document.getElementById('modal-cart');
        const modalCheckout = document.getElementById('modal-checkout');
        const areaLogada = document.getElementById('area-logada');
        const cartCount = document.getElementById('cart-count');
        const cartItems = document.getElementById('cart-items');
        const cartTotal = document.getElementById('cart-total');
        const checkoutTotal = document.getElementById('checkout-total');
        const totalVendasEl = document.getElementById('total-vendas');
        const lucroLiquidoEl = document.getElementById('lucro-liquido');
        const numVendasEl = document.getElementById('num-vendas');

        function addToCart(nome, preco) {
            cart.push({ nome, preco });
            updateCart();
            alert(nome + ' adicionado ao carrinho!');
        }

        function updateCart() {
            cartCount.textContent = cart.length;
            cartItems.innerHTML = cart.map(item => `<div class="cart-item"><span>${item.nome} - R$ ${item.preco}</span></div>`).join('');
            const total = cart.reduce((sum, item) => sum + item.preco, 0);
            cartTotal.textContent = total;
            checkoutTotal.textContent = total;
            document.getElementById('checkout-btn').disabled = cart.length === 0;
        }

        function closeModals() {
            [modalColab, modalCart, modalCheckout].forEach(m => m.classList.remove('show'));
        }

        document.getElementById('view-cart').addEventListener('click', () => {
            modalCart.classList.add('show');
        });

        document.getElementById('checkout-btn').addEventListener('click', () => {
            modalCart.classList.remove('show');
            modalCheckout.classList.add('show');
        });

        document.getElementById('form-checkout').addEventListener('submit', (e) => {
            e.preventDefault();
            const nome = document.getElementById('nome-comprador').value;
            const endereco = document.getElementById('endereco').value;
            const metodo = document.getElementById('metodo-pgto').value;
            const totalCompra = cart.reduce((sum, item) => sum + item.preco, 0);

            // Salvar logs
            cart.forEach(item => {
                const log = {
                    date: new Date().toLocaleString(),
                    product: item.nome,
                    value: 'R$ ' + item.preco,
                    buyer: nome,
                    method: metodo
                };
                saveLog(log);
                totalVendas += item.preco;
                numVendas++;
                lucroL = totalVendas * 0.8;
            });

            updateFinancials();
            cart = [];
            updateCart();
            modalCheckout.classList.remove('show');
            alert('Compra realizada com sucesso! Pagamento processado via ' + metodo);
        });

        function saveLog(log) {
            const logs = JSON.parse(localStorage.getItem('licitatech_logs')) || [];
            logs.push(log);
            localStorage.setItem('licitatech_logs', JSON.stringify(logs));
        }

        function loadLogs() {
            const logs = JSON.parse(localStorage.getItem('licitatech_logs')) || [];
            document.getElementById('logs-table-body').innerHTML = logs.map(log => `<tr><td>${log.date}</td><td>${log.product}</td><td>${log.value}</td><td>${log.buyer}</td><td>${log.method || 'N/A'}</td></tr>`).join('');
        }

        function updateFinancials() {
            totalVendasEl.textContent = totalVendas.toFixed(2);
            lucroLiquidoEl.textContent = (totalVendas * 0.8).toFixed(2);
            numVendasEl.textContent = numVendas;
        }

        document.getElementById('login-colaborador').addEventListener('click', () => modalColab.classList.add('show'));
        document.getElementById('login-dono').addEventListener('click', () => modalDono.classList.add('show'));

        [modalColab, modalDono, modalCart, modalCheckout].forEach(m => {
            m.addEventListener('click', (e) => {
                if (e.target === m) m.classList.remove('show');
            });
        });

        document.getElementById('form-colaborador').addEventListener('submit', (e) => {
            e.preventDefault();
            if (document.getElementById('user-colab').value && document.getElementById('pass-colab').value) {
                modalColab.classList.remove('show');
                areaLogada.style.display = 'block';
                userLogged = document.getElementById('user-colab').value;
                userType = 'colaborador';
                document.getElementById('user-status').textContent = `Logado como colaborador (${userLogged}).`;
                loadLogs();
                updateFinancials();
            }
        });

        document.getElementById('form-dono').addEventListener('submit', (e) => {
            e.preventDefault();
            if (document.getElementById('user-dono').value === 'admin' && document.getElementById('pass-dono').value === '123') {
                modalDono.classList.remove('show');
                areaLogada.style.display = 'block';
                userLogged = document.getElementById('user-dono').value;
                userType = 'dono';
                document.getElementById('user-status').textContent = `Logado como dono (${userLogged}).`;
                loadLogs();
                updateFinancials();
            } else {
                alert('Credenciais inválidas!');
            }
        });

        document.getElementById('logout').addEventListener('click', () => {
            areaLogada.style.display = 'none';
            userLogged = '';
            userType = '';
        });

        loadLogs();
        updateFinancials();
    </script>
